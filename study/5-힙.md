## 프로그래머스 힙 1번 : 더 맵게 
```python
import heapq

def solution(scoville, K):
    heapq.heapify(scoville)
    cnt = 0
    while scoville[0] < K:
        if len(scoville) <= 1:
            return -1
        cnt += 1
        hot1 = heapq.heappop(scoville)
        hot2 = heapq.heappop(scoville)
        mix  = hot1 + hot2 * 2
        heapq.heappush(scoville, mix)

    answer = cnt
    return answer
```

## 프로그래머스 힙 2번 : 디스크 컨트롤러

```python
from collections import deque
import heapq
def solution(jobs):
    n = len(jobs)
    jobs.sort()
    time =0
    answer = 0
    waiting = list()
    
    cnt=0
    
    while cnt < n:
        # 현재 기준 처리할 수 있는 작업을 작업목록에 추가
        # 작업시간 가장 작은것을 바로 뽑아야하니까, 최소 힙에 저장
        i = 0
        while jobs:
            if jobs[i][0] <= time:
                job = jobs.pop(i)
                heapq.heappush(waiting, (job[1], job[0])) # (걸리는시간, 요청온시간)
            else:
                break
        
        # 작업 목록이 있다면, 작업시간이 가장 작은 것 1개 실행
        if waiting:
            job = heapq.heappop(waiting)
            time += job[0]
            answer += time - job[1]
            cnt += 1
        # 작업 목록이 비어있다면, 시간 +1 
        else:
            time += 1
    return answer // n
```


## 프로그래머스 힙 3번 : 이중우선순위큐

```python
import heapq

def solution(operations):
    answer = []
    for operation in operations:
        op, num = operation.split()
        if op == "I":
            heapq.heappush(answer,int(num))
        elif answer and op == "D":
            # 최댓값 삭제
            if int(num) == 1:            
                answer.remove(max(answer))
            # 최솟값 삭제
            else:                      
                heapq.heappop(answer)
    if not answer:  # 없는 경우, [0, 0] 리턴 
        return [0, 0]
    return [max(answer), min(answer)]   # 최댓값, 최솟값 리턴 
```

## Leetcode - Last stone weight

```python
import heapq
class Solution:
    def lastStoneWeight(self, stones: List[int]) -> int:
        # 오름차순
        for i in range(len(stones)):
            stones[i] = stones[i] * -1
        heapq.heapify(stones)
        while len(stones)>1:
            # stone1에 제일 무거운 애 저장 (가장 작은 음수)
            stone1 = heapq.heappop(stones)

            # 두번째로 무거운 애
            if len(stones)==1:
                stone2 = heapq.heappop(stones)

            # 자식 노드 2개 중, 더 무거운 애를 stone2에 저장
            else:
                stone2_1 = heapq.heappop(stones)
                stone2_2 = heapq.heappop(stones)
                
                stone2 = min(stone2_1, stone2_2)
                heapq.heappush(stones, max(stone2_1, stone2_2))

            # 둘이 같으면 부숴지고 다음 루프
            if stone1==stone2:
                continue
            else:
                stone = stone1 - stone2
                heapq.heappush(stones, stone)
        
        
        if stones:
            return -stones[0]
        else:
            return 0
```

## 백준 11286 : 절댓값 힙

```python
import sys
import heapq

heap = []
for _ in range(int(input())):
    num = int(sys.stdin.readline().rstrip())
    if num == 0:
        if heap:
            pop = heapq.heappop(heap)
            print(pop[1])
        else:
            print(0)
    else:
        heapq.heappush(heap, (abs(num), num))
```
